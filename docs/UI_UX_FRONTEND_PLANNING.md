### 6.2 Implementation

```typescript
// lib/utils/zscore-calculator.ts

import { WHO_STANDARDS } from '../constants/zscore-tables';

export function calculateZScore(
  type: 'BBU' | 'TBU' | 'BBTB',
  gender: 'male' | 'female',
  ageMonths: number,
  measurement: number,
  height?: number // untuk BBTB
): number {
  const standard = WHO_STANDARDS[type][gender];
  const reference = standard.find(s => s.age === ageMonths);
  
  if (!reference) throw new Error('Age out of range');
  
  const { median, sd } = reference;
  const zScore = (measurement - median) / sd;
  
  return Math.round(zScore * 100) / 100;
}

export function getStatusBBU(zScore: number): string {
  if (zScore < -3) return 'Berat Badan Sangat Kurang';
  if (zScore < -2) return 'Berat Badan Kurang';
  if (zScore <= 1) return 'Berat Badan Normal';
  return 'Risiko Berat Badan Lebih';
}

export function getStatusTBU(zScore: number): string {
  if (zScore < -3) return 'Sangat Pendek';
  if (zScore < -2) return 'Pendek';
  if (zScore <= 3) return 'Normal';
  return 'Tinggi';
}

export function getStatusBBTB(zScore: number): string {
  if (zScore < -3) return 'Gizi Buruk';
  if (zScore < -2) return 'Gizi Kurang';
  if (zScore <= 1) return 'Gizi Baik';
  if (zScore <= 2) return 'Berisiko Gizi Lebih';
  if (zScore <= 3) return 'Gizi Lebih';
  return 'Obesitas';
}

export function getStatusColor(zScore: number): string {
  if (zScore < -3) return '#F44336'; // Red - Danger
  if (zScore < -2) return '#FF9800'; // Orange - Warning
  if (zScore <= 1) return '#4CAF50'; // Green - Normal
  return '#FFC107'; // Yellow - Attention
}
```
